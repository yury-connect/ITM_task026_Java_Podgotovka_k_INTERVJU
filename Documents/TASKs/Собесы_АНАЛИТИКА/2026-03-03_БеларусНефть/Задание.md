```java
@Component  
public class InvoiceProcessingService {  
  
    @Autowired  
    private final InvoiceRepository invoiceRepository;  
    @Autowired  
    private final PricingService pricingService;  
    @Autowired  
    private final TransactionTemplate transactionTemplate;  
    @Autowired  
    private final KafkaTemplate<String, String> kafkaTemplate;  
  
    public void generateInvoice(Long clientId, List<InvoiceItem> items) {  
        BigDecimal total = pricingService.calculateTotal(items); //вызов внешнего сервиса  
  
        Invoice invoice = new Invoice(clientId, total);  
        invoiceRepository.save(invoice);  
  
        String event = String.format(  
                "{\"invoiceId\":%d,\"clientId\":%d,\"total\":\"%s\"}",  
                invoice.getId(), clientId, total  
        );  
  
        kafkaTemplate.send("invoice-events", clientId.toString(), event);  
    }  
}  
  
class InvoiceItem {  
    private Long id;  
    private String productId;  
    private int quantity;  
  
    public InvoiceItem() {  
    }  
  
    public InvoiceItem(Long id, String productId, int quantity) {  
        this.id = id;  
        this.productId = productId;  
        this.quantity = quantity;  
    }  
  
    public Long getId() {  
        return id;  
    }  
  
    public void setId(Long id) {  
        this.id = id;  
    }  
  
    public String getProductId() {  
        return productId;  
    }  
  
    public void setProductId(String productId) {  
        this.productId = productId;  
    }  
  
    public int getQuantity() {  
        return quantity;  
    }  
  
    public void setQuantity(int quantity) {  
        this.quantity = quantity;  
    }  
}  
  
class Invoice {  
    private Long id;  
    private Long clientId;  
    private BigDecimal total;  
  
    public Invoice(Long clientId, BigDecimal total) {  
        this.clientId = clientId;  
        this.total = total;  
    }  
  
    public Long getId() {  
        return id;  
    }  
  
    public Long getClientId() {  
        return clientId;  
    }  
  
    public BigDecimal getTotal() {  
        return total;  
    }  
}
```
