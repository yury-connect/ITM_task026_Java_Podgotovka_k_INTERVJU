	"context"
	"fmt"
	"github.com/redis/go-redis/v9"

	client := redis.NewClient(&redis.Options{Addr: "localhost:6379"})
	ctx := context.Background()

	// –ó–∞–ø–∏—Å—å –∏ —á—Ç–µ–Ω–∏–µ

	val, err := client.Get(ctx, "server:status").Result()
	fmt.Println(val)  // "online"
}
